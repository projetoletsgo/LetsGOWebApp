# LetsGOPHP
Repositório criado para o desenvolvimento do projeto da disciplina Laboratório de Engenharia de Software do curso Análise e Desenvolvimento de Sistemas da Fatec Taquaritinga

## Create the MySQL Database and Mock
```
CREATE TABLE TB_CIDADE(
	CO_SEQ_CIDADE INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	S_ESTADO VARCHAR(2),
	S_NOME VARCHAR(50)
) ENGINE=InnoDB;

CREATE TABLE TB_USUARIO(
	CO_SEQ_USUARIO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	CO_CIDADE INT(6) UNSIGNED NOT NULL,
	S_NOME VARCHAR(50),
	S_SOBRENOME VARCHAR(50),
	S_EMAIL VARCHAR(30),
	S_SENHA VARCHAR(10),
	IS_EMAIL_CONFIRMADO BOOL DEFAULT 0,
	DT_NASCIMENTO DATE,
	S_RG VARCHAR(9),
	S_CPF VARCHAR(11),
	FOREIGN KEY (CO_CIDADE) REFERENCES TB_CIDADE(CO_SEQ_CIDADE)
) ENGINE=InnoDB;

CREATE TABLE TB_CODIGO(
	CO_SEQ_CODIGO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	S_VALOR VARCHAR(30),
	DT_VALIDADE DATETIME,
	IS_USADO BOOL DEFAULT 0,
	CO_USUARIO INT(6) UNSIGNED NOT NULL,
	S_TELA VARCHAR(30),
	FOREIGN KEY (CO_USUARIO) REFERENCES TB_USUARIO(CO_SEQ_USUARIO)
) ENGINE=InnoDB;

CREATE TABLE TB_TIPO_NOTIFICACAO(
	CO_SEQ_TIPO_NOTIFICACAO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	S_DESCRICAO VARCHAR(20)
) ENGINE=InnoDB;

CREATE TABLE TB_NOTIFICACAO(
	CO_SEQ_NOTIFICACAO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	CO_TIPO_NOTIFICACAO INT(6) UNSIGNED NOT NULL,
	CO_USUARIO INT(6) UNSIGNED NOT NULL,
	S_DESCRICAO VARCHAR(100),
	DT_ALARME DATETIME NOT NULL,
	IS_ATIVA BOOL DEFAULT 1,
	FOREIGN KEY (CO_TIPO_NOTIFICACAO) REFERENCES TB_TIPO_NOTIFICACAO(CO_SEQ_TIPO_NOTIFICACAO),
	FOREIGN KEY (CO_USUARIO) REFERENCES TB_USUARIO(CO_SEQ_USUARIO)
) ENGINE=InnoDB;

CREATE TABLE TB_STATUS_EXCURSAO(
	CO_SEQ_STATUS_EXCURSAO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	S_DESCRICAO VARCHAR(20)
) ENGINE=InnoDB;

CREATE TABLE TB_EXCURSAO(
	CO_SEQ_EXCURSAO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	S_NOME VARCHAR(50),
	CO_USUARIO INT(6) UNSIGNED NOT NULL,
	F_PRECO FLOAT( 10, 2 ) UNSIGNED,
	CO_STATUS_EXCURSAO INT(6) UNSIGNED NOT NULL DEFAULT 1,
	FOREIGN KEY (CO_USUARIO) REFERENCES TB_USUARIO(CO_SEQ_USUARIO),
	FOREIGN KEY (CO_STATUS_EXCURSAO) REFERENCES TB_STATUS_EXCURSAO(CO_SEQ_STATUS_EXCURSAO)
) ENGINE=InnoDB;

CREATE TABLE TB_PARTICIPA_EXCURSAO(
	CO_SEQ_PARTICIPA_EXCURSAO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	CO_USUARIO INT(6) UNSIGNED NOT NULL,
	CO_EXCURSAO INT(6) UNSIGNED NOT NULL,
	FOREIGN KEY (CO_USUARIO) REFERENCES TB_USUARIO(CO_SEQ_USUARIO),
	FOREIGN KEY (CO_EXCURSAO) REFERENCES TB_EXCURSAO(CO_SEQ_EXCURSAO)
) ENGINE=InnoDB;

CREATE TABLE TB_PARADA(
	CO_SEQ_PARADA INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	CO_CIDADE INT(6) UNSIGNED NOT NULL,
	DT_MOMENTO DATETIME,
	S_OBS VARCHAR(200),
	CO_EXCURSAO INT(6) UNSIGNED NOT NULL,
	IS_IDA BOOL DEFAULT 1,
	FOREIGN KEY (CO_EXCURSAO) REFERENCES TB_EXCURSAO(CO_SEQ_EXCURSAO),
	FOREIGN KEY (CO_CIDADE) REFERENCES TB_CIDADE(CO_SEQ_CIDADE)
) ENGINE=InnoDB;

CREATE TABLE TB_CHAT(
	CO_SEQ_CHAT INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	CO_USUARIO INT(6) UNSIGNED,
	CO_EXCURSAO INT(6) UNSIGNED,
	S_NOME VARCHAR(30),
	FOREIGN KEY (CO_EXCURSAO) REFERENCES TB_EXCURSAO(CO_SEQ_EXCURSAO),
	FOREIGN KEY (CO_USUARIO) REFERENCES TB_USUARIO(CO_SEQ_USUARIO)
) ENGINE=InnoDB;

CREATE TABLE TB_CHAT_USUARIO(
	CO_SEQ_CHAT_USUARIO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	CO_CHAT INT(6) UNSIGNED,
	CO_USUARIO INT(6) UNSIGNED,
	FOREIGN KEY (CO_USUARIO) REFERENCES TB_USUARIO(CO_SEQ_USUARIO),
	FOREIGN KEY (CO_CHAT) REFERENCES TB_CHAT(CO_SEQ_CHAT)
) ENGINE=InnoDB;

CREATE TABLE TB_LOG_CHAT(
	CO_SEQ_LOG_CHAT INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	S_TEXTO VARCHAR(100),
	DT_ENVIO DATETIME,
	CO_USUARIO INT(6) UNSIGNED,
	CO_CHAT INT(6) UNSIGNED NOT NULL,
	FOREIGN KEY (CO_CHAT) REFERENCES TB_CHAT(CO_SEQ_CHAT),
	FOREIGN KEY (CO_USUARIO) REFERENCES TB_USUARIO(CO_SEQ_USUARIO)
) ENGINE=InnoDB;

CREATE TABLE TB_VEICULO(
	CO_SEQ_VEICULO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	CO_USUARIO INT(6) UNSIGNED,
	S_MARCA VARCHAR(30),
	S_PLACA VARCHAR(8),
	S_COR VARCHAR(20),
	I_QTD_LUGARES INT(2) NOT NULL,
	FOREIGN KEY (CO_USUARIO) REFERENCES TB_USUARIO(CO_SEQ_USUARIO)
) ENGINE=InnoDB;

CREATE TABLE TB_VEICULO_EXCURSAO(
	CO_SEQ_VEICULO_EXCURSAO INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	CO_VEICULO INT(6) UNSIGNED NOT NULL,
	CO_EXCURSAO INT(6) UNSIGNED NOT NULL,
	FOREIGN KEY (CO_EXCURSAO) REFERENCES TB_EXCURSAO(CO_SEQ_EXCURSAO),
	FOREIGN KEY (CO_VEICULO) REFERENCES TB_VEICULO(CO_SEQ_VEICULO)
) ENGINE=InnoDB;
```
